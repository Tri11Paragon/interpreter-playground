program -> (function | class)*

function -> 'fun' IDENTIFIER '(' function_args ')' scope

function_args = IDENTIFIER | function_args, IDENTIFIER

class -> 'class' IDENTIFIER '{' (variable_decl | function)* '}'

scope -> '{' (scope | statement)* '}' 

variable_decl -> ('const' | 'var') IDENTIFIER (';' | '=' value_expression ';')

array -> ('[' NUMBER ']' | '[' (expression | scope)* ']')

class_construction -> ~IDENTIFIER '{' named_assignment* '}'

named_assignment -> IDENTIFIER '=' value_expression | named_assignment, named_assignment

fun_call -> variable '(' value_expression* ')'

statement -> (variable_decl | control_flow | return | assignment | (fun_call ';'))

assignment -> variable '=' value_expression ';'

return -> 'return' value_expression ';'

value_expression -> (expression | scope | array | class_construction)

control_flow -> "if" (expression | variable_decl | '(' variable_decl ')' all_compare expression) scope ("else" scope | EMPTY) | 
                "while" (expression | variable_decl | '(' variable_decl ')' all_compare expression) scope | 
                "for" for_expression scope

for_expression -> (variable_decl | EMPTY) ';' for_increment ';' 

for_increment -> (expression | (expression ',' for_increment)*)

compare_equality -> ( "!=" | "==" )
compare_relationship ->( ">" | ">=" | "<" | "<=" )
all_compare -> compare_equality | compare_relationship

expression -> equality
equality -> comparison (compare_equality  comparison )*
comparison -> term (compare_relationship  term )*
term -> factor ( ( "-" | "+" ) factor )*
factor -> unary ( ( "/" | "*" ) unary )*
unary -> ( "!" | "-" ) unary | primary
primary -> NUMBER | STRING | "true" | "false" | "nil" | "(" expression ")" | variable | fun_call

variable -> IDENTIFIER | variable '.' variable | variable[NUMBER | variable]
